Tokens CSS

Cole no início do seu CSS ou em um arquivo :root separado.

/* WOW+ Design Tokens */
:root {
  /* Colors */
  --wow-bg:           #272B42;
  --wow-bg-deep:      #1E2138;
  --wow-bg-card:      #2F3350;
  --wow-orange:       #F5A300;
  --wow-orange-dark:  #E07800;
  --wow-orange-light: #FFBF40;
  --wow-white:        #FFFFFF;
  --wow-text-muted:   rgba(255,255,255,0.60);
  --wow-success:      #4BB543;
  --wow-surface:      rgba(255,255,255,0.06);
  --wow-border:       rgba(255,255,255,0.12);

  /* Gradients */
  --wow-grad-logo:    linear-gradient(135deg, #D95F00 0%, #F5A300 100%);
  --wow-grad-orange:  linear-gradient(90deg,  #E07800 0%, #F5A300 100%);
  --wow-grad-bg:      linear-gradient(160deg, #2F3350 0%, #1E2138 100%);

  /* Font */
  --font: 'Poppins', sans-serif;

  /* Type Scale */
  --text-xs:   11px;  --text-sm: 13px;  --text-base: 15px;
  --text-md:   18px;  --text-lg: 22px;  --text-xl:   28px;
  --text-2xl:  36px;  --text-3xl: 48px;  --text-4xl:  64px;

  /* Font Weights */
  --w-regular: 400;  --w-medium: 500;  --w-semibold: 600;
  --w-bold:    700;  --w-extrabold: 800;  --w-black: 900;

  /* Spacing (base-4) */
  --s-1: 4px;  --s-2: 8px;   --s-3: 12px;  --s-4: 16px;
  --s-5: 20px; --s-6: 24px;  --s-8: 32px;  --s-10: 40px;
  --s-12: 48px;--s-16: 64px; --s-20: 80px;

  /* Border Radius */
  --r-sm: 6px;  --r-md: 12px;  --r-lg: 20px;
  --r-xl: 32px; --r-pill: 9999px;

  /* Shadows */
  --shadow-orange: 0 0 24px rgba(245,163,0,0.35);
  --shadow-card:   0 8px 32px rgba(0,0,0,0.40);
}
